QUERY
SELECT h.id AS hID, h.name AS hNAME, h.color AS hColor, h.fur, h.grams AS hGrams, f.id, f.name, f.color, f.fins, f.grams FROM Code301sandboxing.hamsters AS h
LEFT JOIN Code301sandboxing.goldfish AS f
ON h.grams = f.grams
UNION
SELECT h.id AS hID, h.name AS hNAME, h.color AS hColor, h.fur, h.grams AS hGrams, f.id, f.name, f.color, f.fins, f.grams FROM Code301sandboxing.hamsters AS h
RIGHT JOIN Code301sandboxing.goldfish AS f
ON h.grams = f.grams
ORDER BY hGrams;
